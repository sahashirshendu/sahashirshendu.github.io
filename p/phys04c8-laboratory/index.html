! Plots of P3(x), P4(x)
      program legendrepol
      implicit none
      real legendre, x, xmin, xmax, dx
      xmin = -1.0
      xmax = 1.0
      dx = 0.00001
      x = xmin
      open (2, file="legendre.dat")
      do while (x .lt. xmax)
        write (2, *) x, legendre(x, 3), legendre(x, 4)
      x = x + dx
      end do
      close(2)
      end

      function legendre(x, n)
      implicit none
      integer n, i
      real legendre, P(0:n), x
      P(0) = 1
      P(1) = x
      if (n .le. 1) then
        legendre = P(n)
      else
        do i = 1, n - 1
          P(i+1) = ((2*i+1) * x * P(i) - float(i) * P(i-1)) / float(i+1)
        end do
        legendre = P(n)
      end if
      return
      end

      program legendrepol
      implicit none
      integer m, n
      real legendre, x, xmin, xmax, dx, ints, intd, sums, sumd
      xmin = -1.0
      xmax = 1.0
      dx = 0.00001
      x = xmin
      m = 4
      n = 5
      sums = 0
      sumd = 0
      do while (x .lt. xmax)
        sums = sums + legendre(x, n) * legendre(x, n)
        sumd = sumd + legendre(x, m) * legendre(x, n)
      x = x + dx
      end do
      print *, "Integral of P5(x)*P5(x) =", sums * dx
      print *, "Integral of P4(x)*P5(x) =", sumd * dx
      end

      program deflatepoly
      real, allocatable :: a(:), b(:)
      print *, "Enter the order of the polynomial :"
      read *, n
      allocate(a(0:n), b(0:n-1))
      print *, "Enter the coefficients :"
      print *, "(in increasing value of order)"
      read *, a
      print *, "Enter a root :"
      read *, r
      b(n - 1) = a(n)
      do i = 1, n - 1
        b(n - (i + 1)) = a(n - i) + r * b(n - i)
      end do
      print *, "Coefficients of the polynomial =", a
      print *, "Coefficients of the deflated polynomial =", b
      end

      program gelm
      integer n, i, j, k, q, rn
      real k1,k2
      real,allocatable::a(:,:),x(:)
      print*,'Enter n :'
      read*,n
      allocate(a(n,n+1))
      allocate(x(n))
      print*,'Enter the matrix :'
      do i=1,n
        read*,(a(i,j),j=1,n+1)
      end do
      print*,'Matrix ='
      do i=1,n
        print*,(a(i,j),j=1,n+1)
      end do
      do k=1,n-1
        ! Pivoting
        big=abs(a(k,k))
        rn=k
        do m=k+1,n
          if (abs(a(m,k)).gt.big) then
            big=abs(a(m,k))
            rn=m
          end if
        end do
        if (rn.ne.k) then
          do q=k,n+1
            temp=a(k,q)
            a(k,q)=a(rn,q)
            a(rn,q)=temp
          end do
        end if
        ! End of pivoting
        k1=a(k,k)
        do i=k+1,n
          k2=a(i,k)/k1
          do j=k,n+1
            a(i,j)=a(i,j)-k2*a(k,j)
          end do
        end do
      end do
      print*,'Upper triangular matrix ='
      do i=1,n
        print*,(a(i,j),j=1,n+1)
      end do
      x(n)=a(n,n+1)/a(n,n)
      do i=n-1,1,-1
        c=0
        do j=i+1,n
          c=c+a(i,j)*x(j)
        end do
        x(i)=(a(i,n+1)-c)/a(i,i)
      end do
      print*,'Solution ='
      do i=1,n
        print*,x(i)
      end do
      end

      program seidel
      real,allocatable::a(:,:),x(:)
      print*,'Enter n :'
      read*,n
      e=0.00001
      maxit=10000
      allocate(a(n,n+1))
      allocate(x(n))
      print*,'Enter the matrix :'
      do i=1,n
        read*,(a(i,j),j=1,n+1)
      end do
      print*,'Matrix :'
      do i=1,n
        print*,(a(i,j),j=1,n+1)
      end do
      do i=1,n
        x(i)=0
      end do
      print*,'Iterations :'
      do iter=1,maxit
        big=0
        do i=1,n
          sum=0
          do j=1,n
            if (j.ne.i) then
              sum=sum+a(i,j)*x(j)
            end if
          end do
          temp=(a(i,n+1)-sum)/a(i,i)
          relerror=abs((x(i)-temp)/temp)
          if (relerror.gt.big) then
            big=relerror
          end if
          x(i)=temp
        end do
        print*, x
        if(big.le.e) then
          print*,'Converges to a solution'
          print*,'Solution ='
          do i=1,n
            print*,x(i)
          end do
          stop
        end if
      end do
      print*,'Does not converge in maxit iterations'
      print*,'Solution ='
      do i=1,n
        print*,x(i)
      end do
      end
