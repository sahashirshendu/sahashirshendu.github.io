## Plotting Prime Spirals in Polar Coordintes

import numpy as np
import sympy as sp
import matplotlib.pyplot as plt

# plt.style.use("dark_background")


def polar(x):
    return x * np.cos(x), x * np.sin(x)


def pspiral(li):
    li = np.array(list(li))
    x, y = polar(li)
    plt.axis("off")
    plt.scatter(x, y, s=1)
    plt.axis("square")
    plt.show()


# l = int(input("Number from which you want to plot : "))
l = 0
u = int(input("Number upto which you want to plot : "))

primes = sp.primerange(l, u)
pspiral(primes)
